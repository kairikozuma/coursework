% Average filter
n = 5;
aFilt = ones(n)/(n^2);

% Filter the image
campus3filt = uint8(imfilter(double(campus3), aFilt));

% Show image before and after filtering
figure(1);
imshow(campus3);
title('Original image');

figure(2);
imshow(campus3filt);
title('Averaged image');
dim = [.4201, .55729, .090056, .014478];
